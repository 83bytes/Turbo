33 /*
34  * This header defines the basic integral types of the library.
35  *
36  * This library works by convention with template type parameters only, so values of basic types (Such as ints,
37  * booleans, chars, etc) are specified through boxing.
38  * The implementation is free to use value parameters in certain situations, but the user have never use parameters
39  * which are not types (Except in some construction aliases).
40  *
41  * NOTE: Template parameters are disallowed too. For that purpose, the library provides wrappers to pass templates through
42  * boxing (See tml::lazy in "lazy.hpp").
43  *
44  *
45  * This header defines one boxing template for each integral type of the C++ programming language. All templates are defined as
46  * aliases of the template 'tml::integral_constant', which is an alias of 'std::integral_constant (To provide compatibility with the
47  * C++ Standard Library).
48  * The library defines basic types using the following naming conventions:
49  *
50  * - Same name of the C++ type, using CammelCase: This case allows us to use the same common name without having conflicts with keywords
51  * of the language. This is the default naming convention.
52  * Some examples:
53  *
54  * using a = tml::Int<0>; // int a = 0
55  * using b = tml::Bool<true>; // int b = true
56  * using c = tml::UnsignedChar<'a'>; // int c = 'a'
57  *
58  * - Verbose names, lower case, and underscores: This naming convention tries to mimic the naming convention of the Standard Library, convention
59  * followed by the Turbo Metaprogramming Library. Even if using this naming convention is uniform respect the other features of the library,
60  * such names could be a bit verbose and long (Of course, thats really a subjective opinion. Thats why the library defines multiple naming conventions,
61  * to let the user decide what convention is more readable).
62  *
63  * using a = tml::integer<0>; // int a = 0
64  * using b = tml::boolean<true>; // int b = true
65  * using c = tml::unsigned_character<'a'>; // int c = 'a'
66  *
67  * Also prefix-based names for unsigned and long types are provided:
68  *
69  * using d = tml::ulinteger<2>; // unsigned long int d = 0
70  *
71  * - C++ keywords followed with an underscore: Thats the naming convention followed by the Boost::mpl metaprogramming library:
72  *
73  * using a = tml::int_<0>; // int a = 0
74  * using b = tml::bool_<true>; // int b = true
75  * using c = tml::char_<'a'>; // int c = 'a'
76  *
77  * WARNING: CURRENTLY NOT IMPLEMENTED
78  *
79  * Also this header includes "algebra.hpp" and implements algebraic and logical operations for basic types.
80  */

104  /*
105  * Log10 of an integral value
106  * Returns only the integral part of the logarithm, but the algorithm is fast and has O(1) complexity.
107  * Calling log10 with a negative value results in undefined behaviour.
108  */
109  template<typename T , T V>
110  struct log10<tml::integral_constant<T,V>>
111  {
112  static constexpr const auto value = (std::uint64_t)V; //Promote to the wider known unsigned type.
113 
114  static constexpr const T log = (0x0000000000000000ULL <= value && value < 0x000000000000000AULL) ? 0 : //Interval [10^0,10^1)
115  (0x000000000000000AULL <= value && value < 0x0000000000000064ULL) ? 1 : //Interval [10^1,10^2)
116  (0x0000000000000064ULL <= value && value < 0x00000000000003E8ULL) ? 2 : //Interval [10^2,10^3)
117  (0x00000000000003E8ULL <= value && value < 0x0000000000002710ULL) ? 3 : //Interval [10^3,10^4)
118  (0x0000000000002710ULL <= value && value < 0x00000000000186A0ULL) ? 4 : //Interval [10^4,10^5)
119  (0x00000000000186A0ULL <= value && value < 0x00000000000F4240ULL) ? 5 : //Interval [10^5,10^6)
120  (0x00000000000F4240ULL <= value && value < 0x0000000000989680ULL) ? 6 : //Interval [10^6,10^7)
121  (0x0000000000989680ULL <= value && value < 0x0000000005F5E100ULL) ? 7 : //Interval [10^7,10^8)
122  (0x0000000005F5E100ULL <= value && value < 0x000000003B9ACA00ULL) ? 8 : //Interval [10^8,10^9)
123  (0x000000003B9ACA00ULL <= value && value < 0x00000002540BE400ULL) ? 9 : //Interval [10^9,10^10)
124  (0x00000002540BE400ULL <= value && value < 0x000000174876E800ULL) ? 10 : //Interval [10^10,10^11)
125  (0x000000174876E800ULL <= value && value < 0x000000E8D4A51000ULL) ? 11 : //Interval [10^11,10^12)
126  (0x000000E8D4A51000ULL <= value && value < 0x000009184E72A000ULL) ? 12 : //Interval [10^12,10^13)
127  (0x000009184E72A000ULL <= value && value < 0x00005AF3107A4000ULL) ? 13 : //Interval [10^13,10^14)
128  (0x00005AF3107A4000ULL <= value && value < 0x00038D7EA4C68000ULL) ? 14 : //Interval [10^14,10^15)
129  (0x00038D7EA4C68000ULL <= value && value < 0x002386F26FC10000ULL) ? 15 : //Interval [10^15,10^16)
130  (0x002386F26FC10000ULL <= value && value < 0x016345785D8A0000ULL) ? 16 : //Interval [10^16,10^17)
131  (0x016345785D8A0000ULL <= value && value < 0x0DE0B6B3A7640000ULL) ? 17 : //Interval [10^17,10^18)
132  18; //Interval [10^18,2^64 - 1)
133 
134  using result = tml::integral_constant<T,log>;
135  };
