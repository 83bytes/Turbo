41  /*
42  * Internal namespace holds the implementation of the tml::let metafunction.
43  *
44  * tml::let has three parameters:
45  * NAME: The variable (name) the value will be binded with in the expression.
46  * VALUE: The value to be binded.
47  * EXPRESSION: The functional expression which used the NAME variable.
48  * This expression could have any number of parameters.
49  *
50  * The implementation is divided in two phases: High-level and low-level:
51  * - The high-level phase is in charge of taking the let parameters
52  * and passing them to the low-level phase.
53  *
54  * - The low-level phase selects the different substitution actions to be
55  * performed depending on the category of the let parameters.
56  * For example, it covers the cases when the expression is not a functional
57  * expression (Thus there is no NAME to bind into the expression), and general
58  * case when the expression passed is a functional expression.
59  *
60  * The only case which the low-level phase doesn't cover is the case when the
61  * parameters of the functional expression passed matches the binding name. That
62  * case is covered in the high-level phase. In the other cases, the high-level phase
63  * just forwards the parameters to the low-level phase.
64  */
65

94  /*
95  * Special case for lambda bodies expressions.
96  *
97  * Lambda bodies with evaluating expressions such as 'tml::eval<F,_1>' doesn't work because the placeholder is substituted after the
98  * expression evaluation.
99  *
100  * The library provides the template tml::delayed_eval<F,ARGS...> for that purpose: It holds a functional expression reevaluation
101  * with parameters that may be placeholders. When doing let on such template, tml::let substitutes the letted placeholders with its
102  * values, and the tml::delayed_eval template with tml::eval.
103  *
104  * For that purpose, the low_level phase of tml::let is overrided.

113  };

154  /*
155  * Haskell-like let expression
156  *
157  * Binds a value to a named variable in the specified expression.
158  * This metafunction has the following three parameters:
159  * - NAME: The name (variable) to bind the value with.
160  * - VALUE: The value to be binded.
161  * - EXPRESSION: The expression where the substitution will be applied.
162  *
163  * The result of this metafunction is a copy of the passed expression where
164  * the ocurences of the named variable have been substituted with the value.
165  *
166  * NOTE: Explicit "typename ::result" construct is used instead of tml::eval
167  * here because the evaluation process chacks for function signature
168  * (See 'tml::impl::is_function' trait in "functional.hpp"), which instanttiates
169  * the expression. The problem is because the use of variables and/or placeholders
170  * could lead to instantation errors in metafunctions which expect some properties
171  * (Being a function, for example) on its parameters. Of course variables and
172  * and placeholders doesn't has that properties (Are just empty types), so when such
173  * expressions are instantiated with placeholders the instantation fails.
174  * Thats the reason to avoid tml::let here, a context (tml::let) where the usage of
175  * placeholders and variables is common.
176  */
177  template<typename NAME , typename VALUE , typename EXPRESSION>
178  using let = typename impl::let_impl_high<NAME,VALUE,EXPRESSION>::result;

185  /*
186  * Evaluate a multiple-variable let expression and returns the currified unary let equivalent expression.
187  *
188  * A multiple-variable let expression is of the form:
189  *
190  * multi_let<vars...,values...,expression>
191  *
192  * where vars is a set of expression variables, values a set of values to bind, and expression the expression where
193  * to bind the values.
194  * Each variable of the set of variables passed has to have a corresponding one-by-one value in the values set specified,
195  * in other words, the number of values passed should be the same as the number of variables passed.
196  * The binding correspondence is done in order. For example:
197  *
198  * multi_let<x,y,z,a,b,c,function<x,y,z>>
199  *
200  * Binds the value a to the variable x on expression, the value b to the variable y, and the value c to the variable z.
201  * So the result of that multi-variable let expression is:
202  *
203  * function<a,b,c>
204  *
205  * The currifier parses the argumment pack in the way explained above, so the currifier expects an odd number of let parameters
206  * (The last for the expression and one odd set of argumments with the (variable,value) correspondences).
207  *
208  * The parsing is executed as follows:
209  * - The currifier extracts the last argumment as the target expression.
210  * - The rest odd set of argumments is paritiones in two sets, the first corresponding to the set of variables,
211  * and the later corresponding to the set of values.
212  * - Then the currifier generates the expected currified let expression recursively.
213  */
214  template<typename ARGS>
215  struct multi_let_currifier
216  {
217  static_assert( ( ARGS::length - 1 ) > 0 , "A let expression should have at least one variable-value pair to bind." );
218  static_assert( ( ARGS::length - 1 ) % 2 == 0 , "The set of variables and values should be odd (One value per variable)" );
219 
220  using target = tml::lists::back<ARGS>; //Target expression (See parsing description above)
221 
222  /* Just a type with no meaning used as sentinel in some contexts */
223  struct nil {};

226  /*
227  * Metafunction two retrieve the two sets of argumments (Variables and values)
228  * from the original even set of argumments.
229  */
230  template<typename PROCESSED_PARAMS , typename NON_PROCESSED_PARAMS , bool middle>
231  struct split_in_middle;
232 
233  /*
234  * Recursive case: Just iterate to the next element of the argumment list,
235  *
236  * NOTE: The value of middle flag is ( sizeof...(LEFT_LIST) + 1 ) == sizeof...(TAIL)
237  * Which means the processed params list and the non_processed params list has the same length
238  * in the next call.
239  */
240  template<typename... LEFT_LIST , typename HEAD , typename... TAIL>
241  struct split_in_middle<tml::list<LEFT_LIST...> , tml::list<HEAD,TAIL...> , false>
242  {
243  using next_call = split_in_middle<tml::list<LEFT_LIST...,HEAD>,tml::list<TAIL...>,( sizeof...(LEFT_LIST) + 1 ) == sizeof...(TAIL)>;
244 
245  using variables = typename next_call::variables;
246  using values = typename next_call::values;
247  };
248 
249  /*
250  * Base case:
251  */
252  template<typename... LEFT_LIST , typename... RIGHT_LIST>
253  struct split_in_middle<tml::list<LEFT_LIST...>,tml::list<RIGHT_LIST...>,true>
254  {
255  using variables = list<LEFT_LIST...>;
256  using values = list<RIGHT_LIST...>;
257  };
258 
259  using args_without_expression = tml::lists::pop_back<ARGS>;
260 
261  using variables = typename split_in_middle<tml::empty_list,args_without_expression,false>::variables;
262  using values = typename split_in_middle<tml::empty_list,args_without_expression,false>::values;

265  /*
266  * This metafunction gets the set of variables and values and generates a
267  * currified version of the left expression.
268  *
269  * The last optional parameter is only to lead with the explicit two-empty-lists specialization
270  * on the base case (See the comments on the base case).
271  */
272  template<typename VARIABLES , typename VALUES , typename = nil>
273  struct currify;
274 
275  /*
276  * Recursive case: There are more variables to process, generate an enclosing unary let expression
277  * to bind the current variable-value correspondence.
278  */
279  template<typename CURRENT_VAR , typename... VARS_TAIL , typename CURRENT_VAL , typename... VALS_TAIL>
280  struct currify<list<CURRENT_VAR,VARS_TAIL...>,list<CURRENT_VAL,VALS_TAIL...>,nil>
281  {
282  using next_call = currify<list<VARS_TAIL...>,list<VALS_TAIL...>>;
283  using result = tml::let<CURRENT_VAR,CURRENT_VAL,typename next_call::result>;
284  };
285 
286  /*
287  * Base case: There are no more variables, just return the original expression to apply
288  * let of the last variable on it.
289  *
290  * NOTE: The template-partial-specialization parameter is not neccesary. An explicit
291  * specialization for two empty lists should do the work, but explicit template specializations
292  * are not valid on non-global contexts.
293  *
294  * To ride over that the metafunction has an optional unsused parameter which is specified
295  * as a partial-specialization parameter here to solve the problem.
296  */
297  template<typename SENTINEL>
298  struct currify<list<>,list<>,SENTINEL>
299  {
300  using result = target;
301  };
302 
303  /*
304  * The result of the currification process
305  */
306  using result = typename currify<variables,values>::result;
307  };
