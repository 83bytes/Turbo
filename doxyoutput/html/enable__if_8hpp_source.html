50  * This template gets a set of 'std::enable_if' and returns a list with the instantiated

56  /*
57  * This template behaves like a std::enable_if: Produces a ::type member type if a boolean flag is true.
58  */

62  /*
63  * The implementation 'sfinae_all' template takes a set of 'std::enable_if's, a set of their
64  * ::type instantations, and generates a ::type member alias if all 'std::enable_if's are enabled,
65  * counting the number of ::type instantations.
66  */
67  template<typename IFS , typename EXTRACTED_IFS>
68  struct sfinae_all;

70  template<typename... IFS , typename... EXTRACTED_IFS>
71  struct sfinae_all<tml::list<IFS...>,tml::list<EXTRACTED_IFS...>>
72  : public conditional_type<sizeof...(IFS) == sizeof...(EXTRACTED_IFS)>

76  * The implementation 'sfinae_any' template takes a set of 'std::enable_if's, a set of their
77  * ::type instantations, and generates a ::type member alias if any (Almost one) 'std::enable_if' of the passed set is enabled,

90 /*
91  * The following macros define some syntactic sugar for using multiple SFINAE instructions
92  * via sfinae containers. Turbo provides two SFINAE containers:
93  *
94  * - 'tml::impl::sfinae_all': Is enabled when all `std::enable_if`s passed are enabled (Is a logical and).
95  * - 'tml::impl::sfinae_any': Is enabled when at least one `std::enable_if` passed is enabled (Is a logical or).
96  *
97  * In other words, the conditions passed as `std::enable_if`s are conditions/properties which have to be true to instantiate
98  * the template (Enable it) successfully. 'tml::impl::sfinae_all' enables the template if all of the passed conditions/properties
99  * have been met. On the other hand, 'tml::impl::sfinae_any' enables the template if at least one condition/property is met.
100  *
101  * The macros TURBO_SFINAE_X() instantiates the underlying SFINAE container directly, so the cumbersome typename ::type
102  * is not needed to be written by the user. Also, the macro TURBO_FUNCTION_SFINAE_X() declares a prefix "typename = " very convenient
103  * when doing SFINAE on function templates.
104  *
105  * All of them are dessigned to take multiple enable_if conditions. The macros ENABLE_IF() and DISABLE_IF() are shorthands for that enablers.
106  *
107  * Finally, the library provides the macros TURBO_ENABLE_IF() and TURBO_DISABLE_IF() (And its function template equivalents)
108  * for the cases where multiple conditions/properties are not neccesary.
109  */
110 #define TURBO_SFINAE_ALL(...) typename tml::impl::sfinae_all<tml::list<__VA_ARGS__>,tml::impl::extract_ifs<__VA_ARGS__>>::type
111 #define TURBO_SFINAE_ANY(...) typename tml::impl::sfinae_any<tml::impl::extract_ifs<__VA_ARGS__>>::type
112 #define ENABLE_IF(...) std::enable_if<__VA_ARGS__::value>
113 #define DISABLE_IF(...) std::enable_if<!__VA_ARGS__::value>
114 
115 #define TURBO_FUNCTION_SFINAE_ALL(...) typename = TURBO_SFINAE_ALL(__VA_ARGS__)
116 #define TURBO_FUNCTION_SFINAE_ANY(...) typename = TURBO_SFINAE_ANY(__VA_ARGS__)
117 
118 #define TURBO_ENABLE_IF(...) typename ENABLE_IF(__VA_ARGS__)::type
119 #define TURBO_DISABLE_IF(...) typename DISABLE_IF(__VA_ARGS__)::type
120 
121 #define TURBO_ENABLE_FUNCTION_IF(...) typename = TURBO_ENABLE_IF(__VA_ARGS__)
122 #define TURBO_DISABLE_FUNCTION_IF(...) typename = TURBO_DISABLE_IF(__VA_ARGS__)
