32  /*
33  * Represents an unary (One argumment) lambda expression.
34  *
35  * A lambda expression is just a template that holds a variable and a functional
36  * expression (The lambda body) where the variable is used.
37  * The lambda expression is dessigned to act as a functional expression to be evaluated lazily,
38  * when the value of the parameter is specified. Thats why the 'result' member of function entities
39  * is parametrized.
40  *
41  * Because the 'result' member doesn't work like in other function entities (In other function entities returns
42  * the result, in a lambda computes the result), a lambda cannot be considered a classic function entity.
43  * The implementation trait 'tml::impl::is_function' and the 'tml::eval' implementation are overrided to
44  * cover this special behaviour of lambda expressions.
45  */
46  template<typename X , typename BODY>
47  struct lambda
48  {
49  /*
50  * Lambda body evaluation: Substitute the variable of the lambda in the body with
51  * the argumment passed to the lambda and evaluate the resulting expression.
52  */
53  template<typename ARG>
54  using result = tml::eval<tml::let<X,ARG,BODY>>;
55  };
56  }

91  /*
92  * Represents a multiple-variable lambda expression.
93  *
94  * A lambda expression is just a template that holds a set of variables and a functional
95  * expression (The lambda body) where the variable is used.
96  * The lambda expression is dessigned to act as a functional expression to be evaluated lazily,
97  * when the values of the parameters are specified. Thats why the 'result' member of function entities
98  * is parametrized.
99  *
100  * Because the 'result' member doesn't work like in other function entities (In other function entities returns
101  * the result, in a lambda computes the result), a lambda cannot be considered a classic function entity.
102  * The implementation trait 'tml::impl::is_function' and the 'tml::eval' implementation are overrided to
103  * cover this special behaviour of lambda expressions.
104  */
105  template<typename BODY , typename... VARIABLES>
106  struct multi_lambda
107  {
108  template<typename... ARGS>
109  using result = tml::eval<tml::multi_let<VARIABLES...,
110  ARGS...,
111  BODY
112  >
113  >;
114  };

118  /*
119  * Variadic packs must be placed at the end of the set of template argumments,
120  * but tml::multi_lambda takes variables first and the lambda body later.
121  *
122  * This templates places the body parameter to the begin of the tml::impl::multi_lambda
123  * template to allow that template read the pack of variables.
124  *
125  * An extra step (Performed by 'lambda_builder_2' is needed to fill a list with the parameters
126  * of the builder. Using a list is the only way to match the parameters through partial specialization.
127  *
128  * tml::impl::lambda_builder is specialized to return an unary lambda if ony one variable is passed to the lambda builder.
129  * Unary lambdas are preferred because they not currify their let expression, which is a complex process.
130  */

132  struct lambda_builder;
