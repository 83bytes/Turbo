28  /*
29  * Wraps a template to provide lazy template instantation
30  *
31  *
32  * There are situations where a template will be evaluated/instantiated later, but the
33  * use of placeholders is not possible due to the way they are used inside the template. For example:
34  *
35  * template<typename V>
36  * using increment = std::integral_constant<int,V::value + 1>;
37  *
38  * using result = tml::map<increment<_1>,tml::integer_list<1,2,3,4,5>>; //ERROR: tml::placeholders::_1 has no member "value"
39  *
40  * The template tml::lazy is dessigned to wrap a template in a type (Instead of a template parameter), and instantiate
41  * it later with a set of parameters.
42  * This allows the user to pass templates through metafunctions without instantiating them, making the use of placeholders not
43  * necessary. Also lazy template instantation could be used to leave complex template instantations (Instantations which could take a long
44  * time/effort) until the point of its ussage.
45  *
46  * Also tml::lazy allows to specify the instantation parameter at the declaration point, but note that the template will be instanced only
47  * if required explicitly by the user using tml::lazy_instance. For example:
48  *
49  * using lazy = tml::lazy<std::vector,int>; //Here std::vector<int> is not instanced at all.
50  *
51  * using vector = tml::lazy_instance<lazy>; //Here we explicitly instance the wrapped template, std::vector<int> in this example.
52  *
53  * This could be usefull to store potentially ill-formed template instantations. For example:
54  *
55  * template<typename T>
56  * struct foo;
57  *
58  * template<>
59  * struct foo<char>{};
60  *
61  * using ok = tml::lazy_instance<tml::conditional<tml::true_type,
62  * tml::lazy<foo,char>,
63  * tml::lazy<foo,bool>
64  * >
65  * >;
66  *
67  * In the above example, the ill-formed 'foo<bool>' will never be instantiated, because the conditional selects the first branch.
68  *
69  */
70  template<template<typename...> class F , typename... ARGS>
71  struct lazy
72  {};

79  /*
80  * The metafunction lazy_instance takes a wrapped template L and a set of template parameters,
81  * and returns the corresponding template instance.
82  *
83  * This metafunction is specialized to take care of the two usage cases of tml::lazy:
84  * - Template parameters specified at instantation-point: The template is instanced
85  * with the parameters passed to the 'lazy_instance' metafunction.
86  *
87  * - Template parameters specified at declaration point: The parameters passed to
88  * 'lazy_instance' are ignored, the specified at declaration-point are used instead.
89  *
90  * Also the default template works as an identity function, returning the parameter untouched
91  * if its not a tml::lazy instance. This behaviour helps in some contexts.
92  */
93  namespace impl
94  {
95  /*
96  * A lazy instance of an instanced template is the instance itself.
97  *
98  * tml::lazy_instance<int> returns int
99  */
100  template<typename L , typename... ARGS>
101  struct lazy_instance : public tml::function<L>
102  {};
103 
104  /*
105  * A lazy instance of a tml::lazy instance is an instance of the wrapped templates where
106  * the binded parameters come first, and then the passed parameters:
107  *
108  * tml::lazy_instance<tml::lazy<foo>,int> returns foo<int>
109  * tml::lazy_instance<tml::lazy<foo,int>> returns foo<int>
110  * tml::lazy_instance<tml::lazy<foo,int>,int> returns foo<int,int>
111  */
112  template<template<typename...> class F , typename... ARGS , typename... IARGS>
113  struct lazy_instance<tml::lazy<F,ARGS...>,IARGS...> : public tml::function<F<ARGS...,IARGS...>>
114  {};
115  }
