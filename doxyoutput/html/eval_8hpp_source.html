48  * Here we implement the user-side tml::eval<> metafunction.
49  *
50  * The purpose of that metafunction is to evaluate any kind of expression.
51  * That serves to evaluate functions and other functional expressions.

55  /*
56  * The implementation has three parameters:
57  *
58  * - E: The expression to be evaluated.
59  *
60  * - ARGS: evaluate could be used as a high-order metafunction to evaluate a given
61  * function entity with the specified parameters. This variadic pack is that
62  * set of parameters. The result of the evaluation is the result of evaluating
63  * the functional expresion E with the specified ARGS... arguments.
64  * Note that in this case the argumments are evaluated too (Just for the case they are
65  * functional expressions).
66  *
67  * - SFINAE_FALGS: Parameters used to enable/disable certain specializations.
68  *
69  * Of course this metafunction is a function too, so it stores the result of the evaluation in a 'result' member type.
70  */
71  template<typename E , typename ARGS , typename SFINAE_FLAGS = tml::sfinae_return>
72  struct eval : public tml::function<E> //Default case (Just the identity function) for non evaluable expressions
73  {};

87  /* This is the most simple case: There are no evaluation parameters (So the expression could be any
88  * kind of expression, not just a function) BUT the flag says the expression is not a function.

103  /*
104  * This specialization matches the case when the expression passed is a function (In the STL sense, with a 'result' member).
105  * The result of the evalutation is just forwarded to the implementation, to reduce
106  * template instantation depth.
107  *
108  * So the implementation just inherit the function to get its result.
109  *
110  * The parameters are evaluated too (Could be functional/parametrized expressions) to evaluate the entire
111  * expression recursively.
112  */
113  template<template<typename...> class F , typename... ARGS>
114  struct eval<F<ARGS...>,tml::empty_list,
115  TURBO_SFINAE_ALL(
116  DISABLE_IF(tml::overrides_eval<F<ARGS...>>),
117  ENABLE_IF(tml::is_turbo_function<F<ARGS...>>)
118  )
119  >
120  {
121  using result = typename F<typename eval<ARGS,tml::empty_list>::result...>::result;
122  };

217  /*
218  * Metafunction to evaluate expressions.
219  *
220  * The purpose of this metafunction is to evaluate homogeneously any kind of expression.
221  * Also, this could be used as a high-order metafunction to reevaluate an specified functional
222  * expression with a custom set of parameters.
223  *
224  * Note that the parameters of the expression will be ignored during reevaluation.
225  * If your intention is to pass the expression to a high order metafunction,
226  * you have to fill that parameters even if they will not be used during evaluation.
227  * The set of placeholders defined in "placeholders.hpp" could be used for that purpose.
228  * Also the template wrapper 'tml::lazy' defined in "lazy.hpp" could be used to pass a template
229  * without parameters and pass that parameters later when the evaluation is done.
230  *
231  * The metafunction has the following parameters:
232  * - E: The expression to be evaluated.
233  *
234  * - ARGS...: eval could be used as a high-order metafunction to reevaluate a given
235  * function entity with the specified parameters. This variadic pack is that
236  * set of parameters. The result of the evaluation is the result of evaluating
237  * the functional expresion E with the specified ARGS... arguments.
238  * Note that in this case the argumments are evaluated too (Just for the case they are
239  * functional/parametrized expressions).
240  */
241  template<typename EXPRESSION , typename... ARGS>
242  using eval = typename impl::eval<EXPRESSION , tml::list<ARGS...>>::result;

246  /*
247  * Provides delayed evaluation of an expression until its placeholders are substituted by its values.
248  * Is a special case for lambda bodies expressions.
249  *
250  * Lambda bodies with evaluating expressions such as 'tml::eval<F,_1>' doesn't work because the placeholder is substituted after the
251  * expression evaluation.
252  *
253  * The library provides the template tml::delayed_eval<F,ARGS...> for that purpose: It holds a functional expression reevaluation
254  * with parameters that may be placeholders. When doing let on such template, tml::let substitutes the letted placeholders with its
255  * values, and the tml::delayed_eval template with tml::eval.
256  *
257  * NOTE: See the documentation of the specialization of tml::let in "let_expressions.hpp" for more info.
258  */
259  template<typename F , typename... ARGS>
260  struct delayed_eval : public tml::value_chameleon
261  {};
