49  /*
50  * Turbo uses the first quarter (From the LSB side) of the underlying integer number for decimals.
51  * That is, for tml::fsingle, which uses 32 bit integers, the first 8 bits are for decimals and the rest
52  * for the integer part:
53  *
54  * 32 bits
55  * <----------------------------------->
56  * +------------------------+----------+
57  * | Integer | Decimals |
58  * +------------------------+----------+
59  * <-----------------------><---------->
60  * 24 bits 8 bits
61  *
62  * The templates 'tml::impl::decimal_bits' and 'tml::impl::integer_bits' compute the size on bits of a
63  * fixed-point part given an underlying integer type.
64  *
65  * This is the default config. Turbo uses the macro TURBO_DEBUG_FP_DECIMAL_PART to specify the fraction
66  * of bits which are used for decimals, which of course is 4 by default.
67  *
68  * NOTE: Here I miss C++14 variable templates, but I want to make the library compatible with C++11 at least.
69  */
70 
71 #ifndef TURBO_DEBUG_FP_DECIMAL_PART
72 #define TURBO_DEBUG_FP_DECIMAL_PART 4
73 #endif /* TURBO_DEBUG_FP_DECIMAL_PART */
74

123  /*
124  * Aliases for library-defined single and double precision fixed-point types.
125  *
126  * The implementation of fixed-point types is dessigned to work with any underlying integer type, but
127  * the library defines two aliases: One for single precision (32 bit, 24 integer bits and 8 decimal bits) and
128  * one for double precision (64 bit, 48 integer bits and 16 decimal bits).
129  *
130  * Note that type conversions are not implemented, so fixed-point arithmetic is only valid between fixed-point types with
131  * the same underlying integer type. In other words, is not possible to do algebra between tml::fsingle and tml::fdouble.
132  * Depending on the requeriments, the user should choose one fixed-point type.
133  */
134  template<tml::impl::single_t N>
135  using fsingle = tml::fixed_point<tml::impl::single_t,N>;
136 
137  template<tml::impl::double_t N>
138  using fdouble = tml::fixed_point<tml::impl::double_t,N>;
