29 /*
30  * This header declares a set of functional-style algorithms built upon the
31  * high-order metafunctions defined at "high_order_functions.hpp".
32  *
33  * Algorithms which work with sequences are dessigned to work with sequences
34  * represented by lists (tml::list) and pairs of iterators.
35  *
36  * The library provides the same functionality via two naming conventions:
37  * - Functional naming convention: Algorithms have common names used in functional
38  * programming, such as 'tml::map', 'tml::fold', etc.
39  *
40  * - C++ (STL) naming convention: The names mimic those from the Standard Library
41  * <algoritm> header: 'tml::transform', 'tml::accumulate', etc.
42  *
43  * The implementation uses the functional convention by default.
44  */

56  template<typename F , typename STATE , typename BEGIN , typename END , typename ADVANCE>
57  struct apply_for
58  {
59  using next_state = tml::eval<F,STATE,BEGIN>;
60  using next_counter = tml::eval<ADVANCE,BEGIN>;

131  /*
132  * Returns true if any element of a sequence has some property represented by a
133  * boolean predicate P
134  *
135  * The argumments of the function could be:
136  *
137  * List-based sequence:
138  * ====================
139  * - P: The boolean predicate. It should be a unary boolean function enity.
140  * - SEQ: The sequence, represented as a tml::list.
141  *
142  * Iterators-based sequence:
143  * =========================
144  * - P: The boolean predicate. It should be a unary boolean function enity.
145  * - BEGIN: An iterator pointing to the beginning of the input sequence.
146  * - END: An iterator pointing to the end of the input sequence. Note sequences represented
147  * as pairs of iterators are sequences of the interval [BEGIN,END).
148  */
149  template<typename P , typename... SEQ>
150  using any = tml::foldl<tml::lambda<_1,_2 , tml::logical_or<_1,tml::deval<P,_2>>> , tml::false_type , SEQ...>; //Functional (Haskell-like) name)
151 
152  template<typename P , typename... SEQ>
153  using any_of = tml::any<P,SEQ...>; //C++ (STL-ish) name
154 
155  /*
156  * Returns true if all elements of a sequence have some property represented by a
157  * boolean predicate P
158  *
159  * The argumments of the function could be:
160  *
161  * List-based sequence:
162  * ====================
163  * - P: The boolean predicate. It should be a unary boolean function enity.
164  * - SEQ: The sequence, represented as a tml::list.
165  *
166  * Iterators-based sequence:
167  * =========================
168  * - P: The boolean predicate. It should be a unary boolean function enity.
169  * - BEGIN: An iterator pointing to the beginning of the input sequence.
170  * - END: An iterator pointing to the end of the input sequence. Note sequences represented
171  * as pairs of iterators are sequences of the interval [BEGIN,END).
172  */
173  template<typename P , typename... SEQ>

192  /*
193  * Successively applies a function to a value the specified number of times.
194  *
195  * Function argumments:
196  * - F: Binary function. Takes the current value and returns the next. The first parameter is the current value, and the second is the current
197  * value of the counter.
198  * - STATE: Initial value.
199  * - BEGIN , END: The number of times the transformation is applied to the value is represented as a range [BEGIN,END), where BEGIN and END are values.
200  * - ADVANCE: Given the current element of the "number-of-times interval" (See above), computes the next element. By default increases the value by 1.
201  */
202  template<typename F , typename STATE , typename BEGIN , typename END , typename ADVANCE = tml::lambda<tml::placeholders::_1 , tml::add<tml::one<BEGIN>,tml::placeholders::_1>>>
203  using apply_for = typename tml::impl::apply_for<F,STATE,BEGIN,END,ADVANCE>::result;

205  /*
206  * Successively applies a function to a value while a certain condition is met.
207  *
208  * Function argumments:
209  * - F: Unary function. Takes the current value and returns the next.
210  * - STATE: Initial value.
211  * - WHILE: Binary boolean function. Represents the condition to be met. The first parameter is the current value (Value after current function application), and the second the previous one (Value just after
212  * current function application).
213  * This allows to easily write convergence conditions. If you only want a property based on the current value, just ignore the second function param.
214  */
215  template<typename F , typename STATE , typename WHILE>
216  using apply_while = typename tml::impl::apply_while<F,STATE,WHILE>::result;
